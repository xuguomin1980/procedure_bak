F_YF_GETLYR	FUNCTION	1	function f_yf_getlyr(arg_ksdm in varchar2,arg_crkdm in number ,arg_crkdh in varchar2,arg_xh in number,arg_flag in varchar2) return varchar2 is   ls_return  varchar2(50);  begin    begin     If  arg_flag='my' then        IF  instr('+59+61+67+68+','+'||to_char(arg_crkdm)||'+')>0 Then            SELECT   '('||brxm||')'  into ls_return            FROM  MY_CKD           Where  ksdm=arg_ksdm         AND  CRKDM  = arg_crkdm          And  ckdh=arg_crkdh         And  xh=arg_xh         and rownum=1;        End IF;     End IF;          exception           when  no_data_found then          ls_return:='';     end;   return(ls_return); end;
